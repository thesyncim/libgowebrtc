# CMake configuration for building the STUB shim (no libwebrtc required)
# This builds a functional test library for verifying FFI bindings

cmake_minimum_required(VERSION 3.16)
project(libwebrtc_shim_stub VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_POSITION_INDEPENDENT_CODE ON)

# Build shared library
set(BUILD_SHARED_LIBS ON)

# Stub sources
set(SHIM_SOURCES
    shim_stub.cc
)

# Create the shared library
add_library(webrtc_shim ${SHIM_SOURCES})

target_include_directories(webrtc_shim
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}
)

# Set output name
set_target_properties(webrtc_shim PROPERTIES
    OUTPUT_NAME "webrtc_shim"
    VERSION ${PROJECT_VERSION}
    SOVERSION 1
)

# Platform-specific settings
if(APPLE)
    set_target_properties(webrtc_shim PROPERTIES
        SUFFIX ".dylib"
    )
elseif(UNIX)
    set_target_properties(webrtc_shim PROPERTIES
        SUFFIX ".so"
    )
endif()

# Visibility settings
set_target_properties(webrtc_shim PROPERTIES
    CXX_VISIBILITY_PRESET hidden
    VISIBILITY_INLINES_HIDDEN ON
)

message(STATUS "")
message(STATUS "Building STUB shim (no libwebrtc required)")
message(STATUS "  This is for testing FFI bindings only!")
message(STATUS "")
